CREATE TABLE users (
user_id INTEGER PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(100) NOT NULL COMMENT '名稱',
email VARCHAR(255) NOT NULL UNIQUE  COMMENT '信箱,一個信箱只能註冊一次',
